import android.content.Intent import android.provider.Settings import android.widget.Toast import androidx.appcompat.app.AppCompatActivity // داخل الكلاس الخاص بالنشاط class SecurityActivity : AppCompatActivity() { private fun openAdvancedSecuritySettings() { try { // المحاولة الأولى: فتح واجهة تعيين كلمة مرور جديدة مباشرة val intent = Intent(Settings.ACTION_SET_NEW_PASSWORD) startActivity(intent) } catch (e: Exception) { // المحاولة الثانية: في حال فشل الأولى، نفتح إعدادات الأمان العامة try { val intentFallback = Intent(Settings.ACTION_SECURITY_SETTINGS) startActivity(intentFallback) } catch (ex: Exception) { // المحاولة الأخيرة: إبلاغ المستخدم بوجود مشكلة Toast.makeText(this, "تعذر الوصول إلى إعدادات النظام", Toast.LENGTH_LONG).show() } } } }
